Unit 23 - Spatio-temporal parallelization¶

This unit is focused on improving the script created in Unit 22 - Spatio-temporal scripting by processing Sentinel scenes in parallel. Paralelization is performed by ParallelModuleQueue similarly as in Unit 19 - DTM script parallelization, see line 207.

New PyGRASS functionality is introduced on line 136. By MultiModule it is possible to define a list of modules which will work as isolated units not influenced by other processes running parallel. By setting set_temp_region computation region settings will be also isolated from other processes.

The option ncproc on line 43 enables controlling number of processes running in parallel.

 57def cleanup(idx):
 58    Module('g.remove', flags='f', name='mask' + idx, type='raster')
 59    Module('g.remove', flags='f', name='ndvi' + idx, type='raster')

 63def compute(b4, b8, msk, output, idx):

 65    modules = []
 66    modules.append(
 67        Module("g.region",

 73    modules.append(
 74        Module("r.mapcalc",
 75               overwrite = True,
 76               expression = "ndvi{idx} = if(isnull({clouds}), null(), float({b8} - {b4}) / ({b8} + {b4}))".format(
 77                   idx=idx, clouds=msk, b8=b8, b4=b4),
 78               run_ = False)
 79    )

 820.1:0.5:2
 830.5:1:3"""

146def stats(output, date, fd):
147    fd.write('-' * 80)
148    fd.write('\n')
149    fd.write('NDVI class statistics ({0}: {1})'.format(output, date))
150    fd.write('\n')
151    fd.write('-' * 80)
152    fd.write('\n')
153    from subprocess import PIPE
154    ret = Module('v.report', map=output, option='area',
155                 stdout_=PIPE)
156    for line in ret.outputs.stdout.splitlines()[1:]: # skip first line (cat|label|area)
157        # parse line (eg. 1||2712850)
158        data = line.split('|')
159        cat = data[0]
160        area = float(data[-1])
161        fd.write('NDVI class {0}: {1:.1f} ha'.format(cat, area/1e4))
162        fd.write('\n')
163
164    data = vector_db_select(output)
165    for vals in data['values'].values():
166        # unfortunately we need to cast values by float
167        fd.write('NDVI class {0}: {1:.4f} (min) {2:.4f} (max) {3:.4f} (mean)'.format(
168            vals[0], float(vals[2]), float(vals[3]), float(vals[4])))
169        fd.write('\n')

171def main():
172    import grass.temporal as tgis
173
174    tgis.init()
175
176    sp4 = tgis.open_old_stds(options['b4'], 'raster')
177    sp8 = tgis.open_old_stds(options['b8'], 'raster')
178    msk = tgis.open_old_stds(options['mask'], 'raster')
179
180    idx = 1
181    data = []
182    for item in sp4.get_registered_maps(columns='name,start_time'):
183        b4 = item[0]
184        date=item[1]
185        b8 = sp8.get_registered_maps(columns='name',
186                                     where="start_time = '{}'".format(date))[0][0]
187        ms = msk.get_registered_maps(columns='name',
188                                     where="start_time = '{}'".format(date))[0][0]
189        output = '{}_{}'.format(options['basename'], idx)
190        compute(b4, b8, ms, output, str(idx))
191
192        data.append(
193            (output, date)
194        )
195            
196        idx += 1
197
198    queue.wait()
199
200    idx = 1
201    fd = open(options['output'], 'w')
202    for output, date in data:
203        stats(output, date, fd)
204        cleanup(str(idx))
205        idx += 1
206
207    fd.close()

209    return 0
