Unit 30 - PyWPS LST region zones¶

A last (fourth) version of the process will return for input region and dates, LST zones in vector format (GML). The output parameter is defined as ComplexOutput, see lines 26-27. Zones defined on lines 83-86 are created by r.recode (90). Than raster zones are convert to vector by r.to.vect on line 97. Finally vector zones are exported into GML format on line 104 by v.out.ogr. Response content is set on line 110.

  1#!/usr/bin/env python3
  2
  3from subprocess import PIPE
  4from datetime import datetime
  5
  6from grass.pygrass.modules import Module
  7from grass.script import parse_key_val
  8from pywps import Process, LiteralInput, ComplexInput, ComplexOutput, Format
  9
 10
 11class ModisV4(Process):
 12
 13    def __init__(self):
 14        inputs = list()
 15        outputs = list()
 16
 17        inputs.append(ComplexInput('region', 'Input vector region',
 18                    supported_formats=[
 19                        Format('text/xml'), # requires QGIS WPS client
 20                        Format('application/gml+xml')]))
 21        inputs.append(LiteralInput('start', 'Start date (eg. 2023-03-01)',
 22                                   data_type='string'))
 23        inputs.append(LiteralInput('end', 'End date (eg. 2023-03-01)',
 24                                   data_type='string'))
 25
 26        outputs.append(ComplexOutput('zones', 'Output LST zones',
 27                                     supported_formats=[Format('application/gml+xml')])
 28
 29        super(ModisV4, self).__init__(
 30            self._handler,
 31            identifier="modis-v4",
 32            tiotle="Modis process (v4)",
 33            inputs=inputs,
 34            outputs=outputs,
 35            # here you could also specify the GRASS location, for example:
 36            # grass_location="EPSG:5514",
 37            abstract="Computes LST stats for given area and period (limited to Germany and 2023).",
 38            verosion="0.4",
 39            store_supported=True,
 40            status_supported=True)
 41
 42    @staticmethod
 43    def _handler(request, response):
 44        def check_date(date_str):
 45            d = datetime.strptime(date_str, '%Y-%m-%d')
 46            if d.year != 2023:
 47                raise Exception("Only year 2023 allowed")
 48
 49        check_date(request.inputs['start'][0].data)
 50        check_date(request.inputs['end'][0].data)
 51
 52        Module('v.import',
 53                input=request.inputs['region'][0].file,
 54                output='poly')
 55        Module('g.region', vector='poly', align='c_001')
 56        Module('r.mask', vector='poly')
 57
 58        Module("t.rast.series",
 59               overwrite=True,
 60               input="modis_c@PERMANENT",
 61               method="average",
 62               order="start_time",
 63               nprocs=1,
 64               memory=300,
 65               where="start_time > '{}' and start_time < '{}'".format(
 66                   request.inputs["start"][0].data,
 67                   request.inputs["end"][0].data),
 68               output="t_rast_series_out",
 69               file_limit=1000)
 70
 71        m = Module("r.univar",
 72                   flags="g",
 73                   overwrite=True,
 74                   map="t_rast_series_out",
 75                   percentile=90,
 76                   nprocs=1,
 77                   separator="pipe",
 78                   stdout_=PIPE)
 79
 80        stats = parse_key_val(m.outputs.stdout, val_type=float)
 81
 82        output = 'stats'
 83        recode_table = """{min}:{first}:1
 84{first}:{median}:2
 85{median}:{third}:3
 86{third}:{max}:4""".format(
 87    min=stats['min'], first=stats['first_quartile'],
 88    median=stats['median'], third=stats['third_quartile'],
 89    max=stats['max'])                     
 90        Module("r.recode",
 91               overwrite=True,               
 92               input = output,
 93               output = output + '_zones',
 94               rules = "-",
 95               stdin_=recode_table)
 96
 97        Module("r.to.vect",
 98               overwrite=True,
 99               flags = 'sv',
100               input = output + '_zones',
101               output = output + '_zones',
102               type = "area")
103
104        Module('v.out.ogr',
105               overwrite=True,
106               input=output + '_zones',
107               output='zones.gml',
108               format='GML')
109
110        response.outputs['zones'].file = 'zones.gml'
111
112        return response
113
114
115if __name__ == "__main__":
116    from pywps.app.Service import Service
117
118    processes = [ModisV4()]
119    application = Service(processes)

Important

Do not forget to import process as done in Unit 27 and restart demo PyWPS server.

http://localhost:5000/wps?request=Execute&service=WPS&identifier=modis-v4&version=1.0.0&datainputs=start=2023-03-01;end=2023-04-01;region=@xlink:href=http://localhost:5000/static/data/jena_boundary.gpkg

Compare with (output return as reference):

http://localhost:5000/wps?request=Execute&service=WPS&identifier=modis-v4&version=1.0.0&datainputs=start=2023-03-01;end=2023-04-01;region=@xlink:href=http://localhost:5000/static/data/jena_boundary.gpkg&responsedocument=zones=@asReference=true
