Unit 30 - PyWPS LST region zonesÂ¶

A last (fourth) version of the process will return for input region and dates, LST zones in vector format (GML), see line 17. Zones are defined on line 83 by r.recode. Raster classes are convert to vector by r.to.vect on line 96. Finally vector classes are exported into GML format on line 102 by v.out.ogr.

  1import os
  2
  3from pywps import Process, LiteralInput, LiteralOutput, ComplexInput, Format, ComplexOutput
  4
  5__author__ = 'Martin Landa'
  6
  7class ModisV4(Process):
  8    def __init__(self):
  9        inputs = [ComplexInput('region', 'Input vector region',
 10                               supported_formats=[
 11                                   Format('text/xml'), # required by QGIS 2 WPS client
 12                                   Format('application/xml')]),
 13                  LiteralInput('start', 'Start date (eg. 2019-03-01)',
 14                               data_type='string'),
 15                  LiteralInput('end', 'End date (eg. 2019-04-01)',
 16                               data_type='string')]
 17        outputs = [ComplexOutput('zones', 'Output LST zones',
 18                                supported_formats=[Format('application/gml+xml')])
 19        ]
 20
 21        super(ModisV4, self).__init__(
 22            self._handler,
 23            identifier='modis-v4',
 24            version='0.4',
 25            title="Modis process (v4)",
 26            abstract='The process uses the GRASS GIS to compute LST ' \
 27            'statistics for given period in 2019 for Germany',
 28            profile='',
 29            inputs=inputs,
 30            outputs=outputs,
 31            store_supported=True,
 32            status_supported=True,
 33            grass_location="/home/user/grassdata/germany-modis"
 34        )
 35
 36    def check_date(self, date_str):
 37        from datetime import datetime
 38
 39        d = datetime.strptime(date_str, '%Y-%m-%d')
 40        if d.year != 2019:
 41            raise Exception("Only year 2019 allowed")
 42
 43    def _handler(self, request, response):
 44        from subprocess import PIPE
 45        
 46        import grass.script as gs
 47        from grass.pygrass.modules import Module
 48        from grass.exceptions import CalledModuleError
 49        
 50        start = request.inputs['start'][0].data
 51        end = request.inputs['end'][0].data
 52        self.check_date(start)
 53        self.check_date(end)
 54
 55        output = 'modis_{}'.format(os.getpid())
 56        
 57        # be silent
 58        os.environ['GRASS_VERBOSE'] = '0'
 59
 60        Module('v.import',
 61               input=request.inputs['region'][0].file,
 62               output='poly')
 63        Module('g.region', vector='poly', align='c_001')
 64        Module('r.mask', vector='poly')
 65        try:
 66            Module('t.rast.series',
 67                   input='modis_c@PERMANENT',
 68                   output=output,
 69                   method='average',
 70                   where="start_time > '{start}' and start_time < '{end}'".format(
 71                       start=start, end=end
 72            ))
 73        except CalledModuleError:
 74            raise Exception('Unable to compute statistics')
 75
 76        ret = Module('r.univar',
 77                     flags='ge',
 78                     map=output,
 79                     stdout_=PIPE
 80        )
 81        stats = gs.parse_key_val(ret.outputs.stdout)
 82        
 83        Module("r.recode",
 84               input = output,
 85               output = output + '_zones',
 86               rules = "-",
 87               stdin_ = """{min}:{first}:1
 88{first}:{median}:2
 89{median}:{third}:3
 90        {third}:{max}:4""".format(
 91            min=stats['min'], first=stats['first_quartile'],
 92            median=stats['median'], third=stats['third_quartile'],
 93            max=stats['max'])
 94        )
 95
 96        Module("r.to.vect",
 97               flags = 'sv',
 98               input = output + '_zones',
 99               output = output + '_zones',
100               type = "area")
101        
102        Module('v.out.ogr',
103               input=output + '_zones',
104               output='zones.gml',
105               format='GML', overwrite=True)
106
107        response.outputs['zones'].file = 'zones.gml'
108
109        return response

http://localhost:5000/wps?request=Execute&service=WPS&identifier=modis-v4&version=1.0.0&datainputs=start=2019-03-01;end=2019-04-01;region=@xlink:href=http://localhost:5000/static/data/jena_boundary.gpkg&responsedocument=zones=@asReference=true
